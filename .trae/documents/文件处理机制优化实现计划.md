## 📋 计划概述

- 需要文件哈希计算的用户

### 范围

// 修改文件存储结构
private selectedFile: FileData | null = null;

private async selectFileAsInput(): Promise<void> {
    const input = document.createElement('input');
    input.type = 'file';
    input.accept = '*';
    
    input.onchange = async (event) => {
        const file = (event.target as HTMLInputElement).files?.[0];
        if (!file) return;

        try {
            let fileData: FileData;
            
            if (file.type.startsWith('image/')) {
                // 图片文件：读取为ArrayBuffer并存储
                const arrayBuffer = await file.arrayBuffer();
                const bytes = new Uint8Array(arrayBuffer);
                const base64 = this.arrayBufferToBase64(bytes);
                const mimeType = file.type;
                
                fileData = {
                    name: file.name,
                    type: file.type,
                    size: file.size,
                    data: arrayBuffer,  // 保存原始二进制数据
                    isFile: true,
                    mimeType: mimeType
                };
            } else {
                // 其他文件：读取为文本
                const textContent = await file.text();
                
                fileData = {
                    name: file.name,
                    type: file.type,
                    size: file.size,
                    data: textContent,
                    isFile: true,
                    mimeType: file.type
                };
            }

            // 存储文件数据
            this.selectedFile = fileData;

            // 在输入框显示文件引用标记
            const fileRef = this.createFileReference(fileData);
            const inputArea = this.containerEl.querySelector('.codec-input-area') as HTMLTextAreaElement;
            if (inputArea) {
                inputArea.value = fileRef;
                inputArea.dispatchEvent(new Event('input', { bubbles: true }));
                new Notice(`已选中文件: ${file.name}`);
            }
        } catch (error) {
            new Notice(`文件选择失败: ${error instanceof Error ? error.message : '未知错误'}`);
        }
    };

    input.click();
}

// 创建文件引用标记
private createFileReference(fileData: FileData): string {
    return `::FILE_REFERENCE::${JSON.stringify({
        name: fileData.name,
        type: fileData.type,
        size: fileData.size,
        isFile: true
    })}::`;
}

export abstract class BaseOperation implements Operation {
	// ... 现有属性 ...
	
	// 新增：标识是否支持文件数据
	supportsFileData: boolean = false;

	async execute(input: string | FileData, config: Record<string, unknown>): Promise<OperationResult> {
		try {
			// 智能路由：检测输入类型
			const fileData = this.extractFileData(input);
			
			if (fileData && this.supportsFileData) {
				// 支持文件数据的操作，处理文件对象
				const validation = this.validateInput(fileData);
				if (!validation.valid) {
					return {
						success: false,
						data: '',
						error: validation.error || 'Validation failed'
					};
				}
				const result = await this.executeLogic(fileData, config);
				return {
					success: true,
					data: result,
					fileData: fileData
				};
			} else {
				// 普通文本处理
				const stringInput = typeof input === 'string' ? input : JSON.stringify(input);
				const validation = this.validateInput(stringInput);
				if (!validation.valid) {
					return {
						success: false,
						data: '',
						error: validation.error || 'Validation failed'
					};
				}
				const result = await this.executeLogic(stringInput, config);
				return {
					success: true,
					data: result
				};
			}
		} catch (error) {
			return {
				success: false,
				data: '',
				error: error instanceof Error ? error.message : 'Unknown error occurred'
			};
		}
	}

	// 新增：提取文件数据
	protected extractFileData(input: string | FileData): FileData | null {
		// 如果是FileData对象，直接返回
		if (typeof input === 'object' && input !== null && 'isFile' in input) {
			return input as FileData;
		}
		
		// 如果是字符串，检查是否是文件引用
		if (typeof input === 'string' && input.startsWith('::FILE_REFERENCE::')) {
			try {
				const jsonStr = input.replace('::FILE_REFERENCE::', '').replace('::', '');
				return JSON.parse(jsonStr) as FileData;
			} catch (error) {
				return null;
			}
		}
		
		return null;
	}

	// 扩展executeLogic方法签名
	protected async executeLogic(input: string | FileData, config: Record<string, unknown>): Promise<string> {
		// 默认实现：字符串输入处理
		if (typeof input === 'string') {
			return this.executeWithStringInput(input, config);
		} else {
			throw new Error('此操作不支持文件数据');
		}
	}

	// 新增：字符串输入处理（兼容现有逻辑）
	protected async executeWithStringInput(input: string, config: Record<string, unknown>): Promise<string> {
		throw new Error('子类必须实现executeWithStringInput方法');
	}

	// 扩展validateInput方法
	protected validateInput(input: string | FileData): ValidationResult {
		if (typeof input === 'string') {
			return this.validateStringInput(input);
		} else {
			return this.validateFileInput(input);
		}
	}

	// 新增：字符串输入验证（兼容现有逻辑）
	protected validateStringInput(input: string): ValidationResult {
		// 默认实现，子类可以重写
		if (!input || input.trim().length === 0) {
			return { valid: false, error: '输入不能为空' };
		}
		return { valid: true };
	}

	// 新增：文件输入验证
	protected validateFileInput(input: FileData): ValidationResult {
		if (!input || !input.isFile) {
			return { valid: false, error: '无效的文件数据' };
		}
		return { valid: true };
	}
}

```typescript
export class ImageToBase64Operation extends BaseOperation {
	supportsFileData: boolean = true;  // 标识支持文件数据

	// 重写executeLogic方法
	protected async executeLogic(input: string | FileData, config: Record<string, unknown>): Promise<string> {
		try {
			const format = (config.format as string) || 'data-url';

			// 处理文件数据
			if (typeof input === 'object' && input !== null && 'isFile' in input) {
				const fileData = input as FileData;
				
				// 将ArrayBuffer转换为Base64
				if (fileData.data instanceof ArrayBuffer) {
					const bytes = new Uint8Array(fileData.data);
					const base64 = this.arrayBufferToBase64(bytes);
					return this.formatBase64Output(base64, format);
				}
			}

			// 处理Data URL字符串输入
			if (typeof input === 'string' && input.startsWith('data:image/')) {
				const parts = input.split(',');
				if (parts.length >= 2) {
					const base64Part = parts[1];
					return this.formatBase64Output(base64Part, format);
				}
			}

			// 验证纯Base64数据
			this.validateImageBase64(input as string);
			return this.formatBase64Output(input as string, format);
		} catch (error) {
			throw new Error(`图片Base64转换失败: ${error instanceof Error ? error.message : '未知错误'}`);
		}
	}

	private arrayBufferToBase64(buffer: ArrayBuffer): string {
		const bytes = new Uint8Array(buffer);
		let binary = '';
		for (let i = 0; i < bytes.byteLength; i++) {
			binary += String.fromCharCode(bytes[i]);
		}
		return btoa(binary);
	}
}
```

```typescript
export class ReadExifOperation extends BaseOperation {
	supportsFileData: boolean = true;  // 标识支持文件数据

	protected async executeLogic(input: string | FileData, config: Record<string, unknown>): Promise<string> {
		try {
			const EXIF = await import('exif-js');
			
			// 处理文件数据
			if (typeof input === 'object' && input !== null && 'isFile' in input) {
				const fileData = input as FileData;
				
				if (fileData.data instanceof ArrayBuffer) {
					// 直接使用ArrayBuffer创建Image对象
					const image = await this.loadImageFromBuffer(fileData.data);
					const exifData = await this.extractExifData(image, EXIF);
					return this.formatExifAsMarkdown(exifData);
				}
			}

			// 处理Base64字符串输入（兼容现有逻辑）
			if (typeof input === 'string') {
				const imageData = this.cleanInputData(input);
				const image = await this.loadImage(imageData);
				const exifData = await this.extractExifData(image, EXIF);
				return this.formatExifAsMarkdown(exifData);
			}

			throw new Error('不支持的输入格式');
		} catch (error) {
			throw new Error(`EXIF读取失败: ${error instanceof Error ? error.message : '未知错误'}`);
		}
	}

	private async loadImageFromBuffer(buffer: ArrayBuffer): Promise<HTMLImageElement> {
		return new Promise((resolve, reject) => {
			const blob = new Blob([buffer], { type: 'image/jpeg' });
			const url = URL.createObjectURL(blob);
			
			const image = new Image();
			image.onload = () => {
				URL.revokeObjectURL(url);
				resolve(image);
			};
			image.onerror = () => {
				URL.revokeObjectURL(url);
				reject(new Error('图片加载失败'));
			};
			image.src = url;
		});
	}
}
```

**理由**：
- 减少内存占用
- 提高处理效率
